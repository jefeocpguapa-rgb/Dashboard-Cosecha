function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var cb = e && e.parameter && e.parameter.callback; var tipo = e && e.parameter && e.parameter.tipo; if (tipo === "paradas") { var sheetP = ss.getSheetByName("Paradas"); var rows2 = sheetP ? sheetP.getDataRange().getValues() : []; var paradas = []; for (var i = 1; i < rows2.length; i++) { if (rows2[i][0]) paradas.push({ fecha: Utilities.formatDate(new Date(rows2[i][0]), "America/Bogota", "yyyy-MM-dd"), inicio: rows2[i][1], fin: rows2[i][2], minutos: rows2[i][3], obs: rows2[i][4] }); } var r2 = JSON.stringify({paradas: paradas}); if (cb) return ContentService.createTextOutput(cb+"("+r2+")").setMimeType(ContentService.MimeType.JAVASCRIPT); return ContentService.createTextOutput(r2).setMimeType(ContentService.MimeType.JSON); } if (tipo === "parada_nueva") { var sheetP2 = ss.getSheetByName("Paradas"); if (!sheetP2) sheetP2 = ss.insertSheet("Paradas"); var p = e.parameter; sheetP2.appendRow([p.fecha, p.inicio, p.fin, parseInt(p.minutos), p.obs||""]); var r3 = JSON.stringify({ok: true}); if (cb) return ContentService.createTextOutput(cb+"("+r3+")").setMimeType(ContentService.MimeType.JAVASCRIPT); return ContentService.createTextOutput(r3).setMimeType(ContentService.MimeType.JSON); } var sheet = ss.getSheetByName("Hoja1") || ss.getActiveSheet(); var rows = sheet.getDataRange().getValues(); var registros = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) { var fechaStr = Utilities.formatDate(new Date(rows[i][0]), "America/Bogota", "yyyy-MM-dd"); var horaVal = rows[i][1]; var horaStr = ""; if (horaVal instanceof Date) { horaStr = Utilities.formatDate(horaVal, "America/Bogota", "HH:mm"); } else { horaStr = String(horaVal); } registros.push({ fecha: fechaStr, hora: horaStr, bin: rows[i][2], gf: rows[i][3], lote: rows[i][4], finca: rows[i][5], obs: rows[i][6] }); } } var result = JSON.stringify({registros: registros}); if (cb) { var p2 = e.parameter; if (p2.fecha && p2.hora && p2.bin) { sheet.appendRow([p2.fecha, p2.hora, p2.bin, p2.gf||"", p2.lote||"", p2.finca||"", p2.obs||""]); result = JSON.stringify({ok: true}); } return ContentService.createTextOutput(cb+"("+result+")").setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(result).setMimeType(ContentService.MimeType.JSON); }
